
//serialize primitive to char and find size
template<typename T>
struct AsBytes {
	static const char* get(const T& value){
		return reinterpret_cast<const char*>(&value);
	}

	static size_t size(const T&){
		return sizeof(T);
	}
};

//serialize std::string to char and find size 
template<>
struct AsBytes<std::string> {
    static const char* get(const std::string& s) {
        return s.c_str();
    }

    static size_t size(const std::string& s) {
        return s.size();
    }
};
